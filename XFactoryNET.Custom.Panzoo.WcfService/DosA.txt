Private Sub mAvvia()
'Avvia la produzione
'Predispone i dati per il PLC
'specificando nr. silos e qtŕ da estrarre per ogni bilancia
'Elabora solo i materiali in silos.
'I materiali in sacchi (aggiunte manuali) sono visualizzati da altro task.

Dim iBil As Integer
Dim iIndex As Integer
Dim j As Integer
Dim Qtŕ As Integer
Dim Silos As Integer
Dim fEnable As Integer
Dim lWord As Long
Dim sName As String
Dim rsTmp As New Recordset
Dim rsParam As New Recordset
    
    bRunning = True
    bAbort = False
    
    
    ShowDialog = (GetSetting("EasyMenů", "Settings", "ShowErrors", "1") = "1")
    
    SaveSetting "EasyMenů", "Settings", "ShowErrors", IIf(ShowDialog, "1", "0")
    
    On Error GoTo OnRkErr
    
    rsParam.Open "T_Parametri", gConn, adOpenDynamic, adLockOptimistic
    rsParam.Find "IdParametro='$IdArch'"
    If rsParam.EOF Then
        rsParam.AddNew
        rsParam.Fields!IdParametro = "$IdArch"
        rsParam.Fields!Default = 1
        rsParam.Fields!Descrizione = "Progressivo archiviazione"
        gIdArch = 1
        rsParam.Update
    Else
        gIdArch = rsParam.Fields!Default
    End If
    
    rsParam.Close
    
    mrsBil.Open "SELECT * FROM V_Bilance WHERE IdLav='" & mIDLav & "' ORDER BY Name", gConn, adOpenStatic, adLockReadOnly
    
    mrsProd.Open "SELECT * FROM V_Produzioni WHERE IdArch=" & gIdArch & " AND IdLav = '" & mIDLav & "' AND Stato=0 ORDER BY IdProd", gConn, adOpenDynamic, adLockOptimistic
    If Not mrsProd.EOF Then
        mrsProd.MoveFirst
        mIdProd = mrsProd.Fields!IdProd
        mNrMisc = mrsProd.Fields!NrMisc
        iStopMisc = mNrMisc + 1
        mIdMat = mrsProd.Fields!IdMateriale
        mIdArt = mrsProd.Fields!IdArticolo
        bGranMenů = mrsProd.Fields!GranMenů
        bHidden = mrsProd.Fields!Hidden              'Produzione "nascosta"
        
        iMix = 0
        iDS1 = 0
        
        rsParam.Open "Select * FROM T_UsoParametri WHERE IDArch = " & gIdArch & " And IDProd = " & mIdProd, gConn, adOpenDynamic, adLockReadOnly
        If Not rsParam.EOF Then
            rsParam.Find "IDParametro = '$Mixer'"
            If Not rsParam.EOF Then
                iMix = rsParam.Fields!Valore
            End If
            rsParam.MoveFirst
            rsParam.Find "IDParametro = '$DS1'"
            If Not rsParam.EOF Then
                iDS1 = rsParam.Fields!Valore
            End If
        End If
        rsParam.Close
        
        ShowMsg "Avvio produzione nr. " & mIdProd
        
        aDummy(1) = 0
        rc = mRK512.RK512Get(rkData, 39, 10, 1, aDummy)
        If rc <> 0 Then            'SetPoint
            App.LogEvent "Errore lettura DB39.DW20 (abilitazione PC) rc=" & rc, vbLogEventTypeError
            GoTo rkErr
        End If
        If (aDummy(1) = 0) Then
            'PC non abilitato
            ShowMsg "PC non abilitato"
            GoTo rkErr
        End If
        
        mCurrMisc = 1
        
        'Scrivo silos e qtŕ
        'con NrMisc=0 sono indicate le qtŕ teoriche
        'Controllo la presenza di additivi manuali da aggiungere direttamente in mixer (sacchettino)
        rsTmp.Open "SELECT * FROM V_DettProduzioniMaga WHERE IdArch=" & gIdArch & " AND IdMod='ADD' AND NrMisc=0 AND IdProd=" & mIdProd, gConn, adOpenStatic, adLockReadOnly
        bAdditivi = Not rsTmp.EOF
        rsTmp.Close
        
        mrsDett.Open "SELECT * FROM V_DettProduzioniSilos WHERE IdArch = " & gIdArch & " AND NrMisc=0 AND IdProd = " & mIdProd & " ORDER BY Qtŕ DESC", gConn, adOpenStatic, adLockReadOnly
        fEnable = 0
        For iBil = 1 To cNRBIL
            lStart(iBil) = 0
            sPeso(iBil) = 0
            iProgr(iBil) = 1
            sEstratto(iBil) = 0
            mrsDett.Filter = "IdApparato='" & sBilName(iBil) & "'"
            With mrsDett
                For j = 1 To cNRINGR
                    If Not .EOF Then
                        fEnable = fEnable Or (2 ^ (iBil - 1))
                        bFineBil(iBil) = False
                        lStart(iBil) = 1
                        Qtŕ = CLng(.Fields!Qtŕ / kCorr(iBil))
                        Silos = CLng(.Fields!IdCella)
                        sQtŕ(iBil, j) = .Fields!Qtŕ                 'salvo la qtŕ teorica
                        sIdMat(iBil, j) = .Fields!IdMateriale
                        sMod(iBil, j) = .Fields!IdMod
                        sToll(iBil, j) = .Fields!Toll
                        sIdCella(iBil, j) = .Fields!IdCella
                        .MoveNext
                    Else
                        Qtŕ = 0
                        Silos = 0
                        sQtŕ(iBil, j) = 0
                        sIdMat(iBil, j) = vbNullString
                        sMod(iBil, j) = vbNullString
                        sToll(iBil, j) = 0
                        sIdCella(iBil, j) = vbNullString
                    End If
                    aData(j * 2 - 1) = Silos
                    aData(j * 2) = Qtŕ
                    If sIdCella(iBil, j) <> vbNullString And sQtŕ(iBil, j) <> 0 Then
                        ShowMsg "Produzione Nr. " & mIdProd & _
                                " Bilancia: " & sBilName(iBil) & _
                                " Silos: " & sIdCella(iBil, j) & _
                                " Q.tŕ: " & sQtŕ(iBil, j)
                    End If
                Next j
            End With
            If Not mSimul And lStart(iBil) = 1 Then
                rc = mRK512.RK512Put(rkData, 50 + iBil, 51, cNRINGR * 2, aData)
                If rc <> 0 Then            'SetPoint
                    App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW102 (Qtŕ,Silos) rc=" & rc, vbLogEventTypeError
                    GoTo rkErr
                End If
                aDummy(1) = CInt(NrMisc)
                '03/08 rc = mRK512.RK512Put(rkData, 50 + iBil, 21, 1, aDummy)
                rc = mRK512.RK512Put(rkData, 50 + iBil, 23, 1, aDummy)
                If rc <> 0 Then             'NrMisc
                    'App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW21 (NrMisc) rc=" & rc, vbLogEventTypeError
                    App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW46 (NrMisc) rc=" & rc, vbLogEventTypeError
                    GoTo rkErr
                End If
                aDummy(1) = Val(mrsProd.Fields!IdMateriale) 'Conversione in integer
                '03/08 rc = mRK512.RK512Put(rkData, 50 + iBil, 22, 1, aDummy)
                rc = mRK512.RK512Put(rkData, 50 + iBil, 24, 1, aDummy)
                If rc <> 0 Then             'NrForm
                    'App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW22 (NrFormula) rc=" & rc, vbLogEventTypeError
                    App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW48 (NrFormula) rc=" & rc, vbLogEventTypeError
                    GoTo rkErr
                End If
                '03/08
                aDummy(1) = mIdProd        'Nr Lotto (progressivo del turno)
                rc = mRK512.RK512Put(rkData, 50 + iBil, 22, 1, aDummy)
                If rc <> 0 Then
                    App.LogEvent "Errore scrittura DB" & 50 + iBil & ".DW44 (NrLotto) rc=" & rc, vbLogEventTypeError
                    GoTo rkErr
                End If
                
            End If
            mrsBil.MoveNext
        Next iBil
        mrsDett.Close
            
        ShowMsg "Flag Enable = &H" & Hex$(fEnable)
        
        If Not mSimul Then
            '03/08 aDummy(1) = IIf(bGranMenů, 61, 0)
            'rc = mRK512.RK512Put(rkData, 39, 19, 1, aDummy)
            'If rc <> 0 Then            'GranMenů
                'App.LogEvent "Errore scrittura DB39.DW19 (FlagGranMenů) rc=" & rc, vbLogEventTypeError
                'GoTo rkErr
            'End If
            'aDummy(1) = iMix
            'rc = mRK512.RK512Put(rkData, 39, 40, 1, aDummy)
            'If rc <> 0 Then            'MIX1/MIX2
                'App.LogEvent "Errore scrittura DB39.DW40 (MIX1/MIX2) rc=" & rc, vbLogEventTypeError
                'GoTo rkErr
            'End If
            
            aDummy(1) = 0
            If (bGranMenů) Then
                aDummy(1) = &H100
            End If
            aDummy(1) = aDummy(1) Or (iMix)
            rc = mRK512.RK512Put(rkData, 39, 15, 1, aDummy)
            If rc <> 0 Then            'GranMenů | Mixer
                App.LogEvent "Errore scrittura DB39.DW30 (FlagGranMenů+Mixer) rc=" & rc, vbLogEventTypeError
                GoTo rkErr
            End If
            
            '03/08
            'aDummy(1) = iDS1
            'rc = mRK512.RK512Put(rkData, 39, 41, 1, aDummy)
            'If rc <> 0 Then            'DS1
                'App.LogEvent "Errore scrittura DB39.DW41 (DS1) rc=" & rc, vbLogEventTypeError
                'GoTo rkErr
            'End If
            
            'aDummy(1) = IIf(bAdditivi, 1, 0)
            'rc = mRK512.RK512Put(rkData, 39, 17, 1, aDummy)
            'If rc <> 0 Then      'Additivi
                'App.LogEvent "Errore scrittura DB39.DW17 (FlagAdditivi) rc=" & rc, vbLogEventTypeError
                'GoTo rkErr
            'End If
            
            aDummy(1) = IIf(bAdditivi, 2, 0)
            aDummy(1) = aDummy(1) Or ((iDS1 - 1) * 4)
            rc = mRK512.RK512Put(rkData, 39, 19, 1, aDummy)
            If rc <> 0 Then            'Additivi | DS1
                App.LogEvent "Errore scrittura DB39.DW38 (Additivi + DS1) rc=" & rc, vbLogEventTypeError
                GoTo rkErr
            End If
            
           
            sName = Trim(mrsProd.Fields!Descrizione)
            If Len(sName) < 24 Then sName = sName & Space(24 - Len(sName))
            sName = Left(sName, 24)
            For j = 1 To 12
                lWord = CLng(AscB(Mid(sName, j * 2 - 1, 1))) * CLng(256) + CLng(AscB(Mid(sName, j * 2, 1)))
                If (lWord And &H8000) <> 0 Then
                    lWord = lWord - &H10000
                End If
                aData(j) = CInt(lWord)
            Next j
            'rc = mRK512.RK512Put(rkData, 39, 20, 12, aData)
            rc = mRK512.RK512Put(rkData, 39, 51, 12, aData)
            If rc <> 0 Then 'Codice formula
                App.LogEvent "Errore scrittura DB39.DW102 (Codice formula) rc=" & rc, vbLogEventTypeError
                GoTo rkErr
            End If
            
            For iBil = 1 To NrBil
                If Not mSimul Then
                    aDummy(1) = 0
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 4, 1, aDummy)
                    If rc <> 0 Then 'Reset
                        App.LogEvent "Errore scrittura DB109.DW" & 10 * iBil + 4 & " (FlagReset) rc=" & rc, vbLogEventTypeError
                        GoTo rkErr
                    End If
                        
                    aDummy(1) = lStart(iBil)
                    '03/08 rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 1, 1, aDummy)
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil, 1, aDummy)
                    If rc <> 0 Then            'Start
                        App.LogEvent "Errore scrittura DB109.DW" & 2 * (10 * iBil) & " (FlagStart) rc=" & rc, vbLogEventTypeError
                        GoTo rkErr
                    End If
                    aDummy(1) = 0
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 2, 1, aDummy)
                    If rc <> 0 Then                'Fine
                        App.LogEvent "Errore scrittura DB109.DW" & 2 * (10 * iBil + 2) & " (FlagFine) rc=" & rc, vbLogEventTypeError
                        GoTo rkErr
                    End If
                End If
            Next
            
            aDummy(1) = fEnable
            rc = mRK512.RK512Put(rkData, 109, 2, 1, aDummy)
            If rc <> 0 Then            'Enable
                App.LogEvent "Errore scrittura DB109.DW4 (FlagEnable) rc=" & rc, vbLogEventTypeError
                GoTo rkErr
            End If
            aDummy(1) = 1
            rc = mRK512.RK512Put(rkData, 109, 1, 1, aDummy)
            If rc <> 0 Then            'EOT
                App.LogEvent "Errore scrittura DB109.DW2 (EOT) rc=" & rc, vbLogEventTypeError
                GoTo rkErr
            End If
        End If
        
        mrsProd.Fields!Stato = 1        'In produzione
        mrsProd.Update
        
        RaiseEvent InizioProduzione(mIdProd)
        
        RaiseEvent InizioMiscelata(mIdProd, mCurrMisc)
        
        For iBil = 1 To cNRBIL
            If lStart(iBil) = 1 Then RaiseEvent InizioDosaggio(mIdProd, mCurrMisc, sBilName(iBil))
        Next
        
        mrsGiac.Open "T_Silos", gConn, adOpenDynamic, adLockOptimistic
        mrsDettProd.Open "T_DettProduzioni", gConn, adOpenDynamic, adLockOptimistic
        mScanTimer.Enabled = True
    
    Else
        bRunning = False
    End If

exSub:
    mrsBil.Close
    mrsProd.Close
    Exit Sub
    
rkErr:
    ShowMsg "Errore Protocollo rk512"
    mrsProd.Fields!Stato = -1
    mrsProd.Update
    If mrsDett.State = adStateOpen Then mrsDett.Close
    bRunning = False
    mIdProd = 0
    mNrMisc = 0
    mCurrMisc = 0
    mTimer.Enabled = True
    GoTo exSub
    
OnRkErr:
    ShowMsg "Errore rk512: " & Err.Description
    Resume Next
End Sub

Private Sub UpdStato(IdProd As Long, Stato As Integer)
'Aggiorna lo stato della produzione in corso
'Ricalcola il numero di miscelate e la qtŕ totale che possono
'essere cambiate a seguito di stop o abort.

Dim rsTmp As New Recordset

    rsTmp.Open "T_Produzioni", gConn, adOpenDynamic, adLockOptimistic
    rsTmp.Filter = "IdArch=" & gIdArch & " AND IdProd = " & IdProd
    If Not rsTmp.EOF Then
        rsTmp.Fields!Stato = Stato
        rsTmp.Fields!Qtŕ = rsTmp.Fields!Qtŕ / rsTmp.Fields!NrMisc * (mCurrMisc - 1)
        rsTmp.Fields!NrMisc = mCurrMisc - 1
        rsTmp.Update
    End If
    rsTmp.Close

End Sub

Private Sub mScanTimer_Timer()
Dim iBil As Integer
Dim bFine As Boolean
Dim j As Integer
Dim lQtŕ As Single
Static sCount As Integer

'Verifica periodicamente lo stato delle bilance
'Per le bilance avviate legge
' il flag di fine dosaggio
' il peso sulla bilancia
' il nr. progressivo dell'ingrediente in corso di estrazione
' la qtŕ estratta dell'ingrediente attuale
'Se la bilancia ha terminato allora legge i pesi estratti
'Se č richiesto l'abort allora setta a zero i pesi estratti (altrimenti leggerebbe i pesi della miscelata precedente)




    mScanTimer.Enabled = False
    
    On Error GoTo OnRkErr
    
    bFine = True
    
    With mrsDettProd
        For iBil = 1 To cNRBIL
            If Not bFineBil(iBil) And lStart(iBil) = 1 Then
                iStato(iBil) = 0
                If Not mSimul Then
                    rc = mRK512.RK512Get(rkData, 109, 10 * iBil + 2, 1, aDummy)
                    If rc <> 0 Then
                        LogEvent "IdProd= " & mIdProd & " CurMisc=" & CurrMisc & " Errore lettura DB109.DW" & 2 * (10 * iBil + 2) & " (Flag fine dosaggio) rc=" & rc, vbLogEventTypeError
                        aDummy(1) = 0
                    End If
                    iStato(iBil) = aDummy(1)
                    rc = mRK512.RK512Get(rkData, 50 + iBil, 1, 1, aDummy)
                    If rc <> 0 Then
                        LogEvent "IdProd= " & mIdProd & " CurMisc=" & CurrMisc & " Errore lettura DB" & 50 + iBil & ".DW2 (peso bilancia) rc=" & rc, vbLogEventTypeError
                    End If
                    sPeso(iBil) = aDummy(1)
                    'rc = mRK512.RK512Get(rkData, 50 + iBil, 6, 1, aDummy)
                    rc = mRK512.RK512Get(rkData, 50 + iBil, 4, 1, aDummy)
                    If rc <> 0 Then
                        LogEvent "IdProd= " & mIdProd & " CurMisc=" & CurrMisc & " Errore lettura DB" & 50 + iBil & ".DW8 (Nr. progr. ingrediente) rc=" & rc, vbLogEventTypeError
                    End If
                    iProgr(iBil) = aDummy(1) And &HFF   'byte basso
                    rc = mRK512.RK512Get(rkData, 50 + iBil, 16, 1, aDummy)
                    If rc <> 0 Then
                        LogEvent "IdProd= " & mIdProd & " CurMisc=" & CurrMisc & " Errore lettura DB" & 50 + iBil & ".DW32 (Qtŕ estratta) rc=" & rc, vbLogEventTypeError
                    End If
                    sEstratto(iBil) = aDummy(1)
                ElseIf lStart(iBil) = 1 Then
                    sPeso(iBil) = sPeso(iBil) + sQtŕ(iBil, iProgr(iBil)) * (1 + Rnd / 10)
                    iProgr(iBil) = iProgr(iBil) + 1
                    sEstratto(iBil) = 0
                    If sQtŕ(iBil, iProgr(iBil)) = 0 Then
                        iStato(iBil) = 1
                        sPeso(iBil) = 0
                        iProgr(iBil) = 1
                    End If
                End If
                If (lStart(iBil) = 1 And iStato(iBil) = 1) Or bAbort Then
                    If Not mSimul Then
                        rc = mRK512.RK512Get(rkData, 50 + iBil, 101, cNRINGR, aData)
                        If rc <> 0 Then
                            LogEvent "IdProd= " & mIdProd & " CurMisc=" & CurrMisc & " Errore lettura DB" & 50 + iBil & ".DW202 (pesi estratti) rc=" & rc, vbLogEventTypeError
                            'In caso di errore di comunicazione prendo i valori teorici
                            For j = 1 To cNRINGR
                                aData(j) = sQtŕ(iBil, j) / kCorr(iBil)
                            Next
                        End If
                    End If
                    bFineBil(iBil) = True
                    If bAbort Then
                        ShowMsg "Produzione Nr. " & mIdProd & _
                                " Bilancia: " & sBilName(iBil) & ": Abort dosaggio"
                    Else
                        ShowMsg "Produzione Nr. " & mIdProd & _
                                " Bilancia: " & sBilName(iBil) & ": Fine dosaggio"
                    End If
                    'Fine dosaggio bilancia iBil
                    For j = 1 To cNRINGR
                        If sIdMat(iBil, j) <> vbNullString Then
Retry:
                            If Not mSimul Then
                                If bAbort Then
                                    lQtŕ = 0
                                Else
                                    lQtŕ = aData(j)
                                End If
                            Else
                                lQtŕ = sQtŕ(iBil, j) * 0.95 / kCorr(iBil)
                            End If
                            lQtŕ = lQtŕ * kCorr(iBil)       'in kg
                            If Abs(lQtŕ - sQtŕ(iBil, j)) / sQtŕ(iBil, j) * 100 > sToll(iBil, j) Then
                                ShowMsg "Bilancia: " & sBilName(iBil) & _
                                        " Silos: " & sIdCella(iBil, j) & _
                                        " Q.tŕ Teorica: " & sQtŕ(iBil, j) & _
                                        " Q.tŕ Estratta: " & lQtŕ
                                If sToll(iBil, j) <> 0 Then
                                    If ShowDialog Then
                                        Load frmDialog
                                        frmDialog.lblMsg = "Estrazione da bilancia " & _
                                                         sBilName(iBil) & " fuori tolleranza" & vbCrLf & vbCrLf & _
                                                         "Ingrediente: " & sIdMat(iBil, j) & vbCrLf & _
                                                         "Silos: " & sIdCella(iBil, j) & vbCrLf & _
                                                         "Q.tŕ teorica: " & sQtŕ(iBil, j) & vbCrLf & _
                                                         "Q.tŕ estratta: " & lQtŕ
                                        frmDialog.Show vbModal
                                        
    Dim bTry As Boolean
                                        bTry = frmDialog.Retry
                                        Unload frmDialog
                                        If bTry Then
                                            rc = mRK512.RK512Get(rkData, 50 + iBil, 100 + j, 1, aDummy)
                                            aData(j) = aDummy(1)
                                            GoTo Retry
                                        End If
                                    End If
                                End If
                            End If
                            'Scrive record dettaglio produzione
                            .AddNew
                            .Fields!IdArch = gIdArch
                            .Fields!IdProd = mIdProd
                            .Fields!NrMisc = mCurrMisc
                            .Fields!QtŕEstr = lQtŕ
                            .Fields!Qtŕ = sQtŕ(iBil, j)                 'qtŕ teoriche
                            .Fields!IdMateriale = sIdMat(iBil, j)
                            .Fields!Toll = sToll(iBil, j)
                            .Fields!IdCella = sIdCella(iBil, j)
                            .Fields!IdMod = sMod(iBil, j)
                            .Update
                            'Decremento giacenza siloss
                            If Not mrsGiac.BOF Then mrsGiac.MoveFirst
                            mrsGiac.Find "Name='" & sIdCella(iBil, j) & "'"
                            If Not mrsGiac.EOF Then
                                mrsGiac.Fields!Qtŕ = mrsGiac.Fields!Qtŕ - lQtŕ
                                If mrsGiac.Fields!Qtŕ <= 0 Then
                                    mrsGiac.Fields!Qtŕ = 0
                                    mrsGiac.Fields!IdMateriale = Null
                                    mrsGiac.Fields!IdProd = Null
                                    mrsGiac.Fields!IdArch = Null
                                End If
                                mrsGiac.Update
                            End If
                        End If
                    Next j
                    RaiseEvent FineDosaggio(mIdProd, mCurrMisc, sBilName(iBil))
                ElseIf lStart(iBil) = 1 Then
                    bFine = False
                End If
            End If
        Next
    End With
    
    If bFine Then
        ShowMsg "Produzione nr. " & mIdProd & ": Fine miscelata nr." & mCurrMisc
        'Aggiorno cella di destinazione
Dim rsTmp As New Recordset
Dim sql As String

        'Calcolo le quantitŕ effettivamente estratte
        sql = "SELECT Sum(QtŕEstr) as QtŕTot FROM V_DettProduzioni WHERE IdArch=" & gIdArch & " AND IdProd=" & mIdProd & _
            " AND (NrMisc=" & mCurrMisc & ")"
        rsTmp.Open sql, gConn, adOpenStatic, adLockReadOnly
        lQtŕ = rsTmp.Fields!QtŕTot
        rsTmp.Close
        If Not bHidden Then
            'Qui bisognerebbe chiedere conferma della cella di destinazione ...
            rsTmp.Open "SELECT * FROM T_DestinazioneProduzioni WHERE IdArch = " & gIdArch & " AND IdProd=" & mIdProd & " AND NrMisc=" & mCurrMisc, gConn, adOpenStatic, adLockReadOnly
            If Not rsTmp.EOF Then
                mDest = rsTmp.Fields!Dest
            End If
            If Not mrsGiac.BOF Then mrsGiac.MoveFirst
            mrsGiac.Find "Name='" & mDest & "'"
            If Not mrsGiac.EOF Then
                mrsGiac.Fields!Qtŕ = mrsGiac.Fields!Qtŕ + lQtŕ
                mrsGiac.Fields!IdMateriale = mIdMat
                mrsGiac.Fields!IdArticolo = mIdArt
                mrsGiac.Fields!IdProd = mIdProd
                mrsGiac.Fields!IdArch = gIdArch
                mrsGiac.Update
            End If
        End If
        mCurrMisc = mCurrMisc + 1
        RaiseEvent FineMiscelata(mIdProd, mCurrMisc - 1)
        
        If Not bAbort And mCurrMisc <= mNrMisc And mCurrMisc <= iStopMisc Then
            ShowMsg "Produzione nr. " & mIdProd & ": Inizio miscelata nr." & mCurrMisc
            RaiseEvent InizioMiscelata(mIdProd, mCurrMisc)
            For iBil = 1 To cNRBIL
                If lStart(iBil) = 1 Then
                    bFineBil(iBil) = False
                    aDummy(1) = 0
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 4, 1, aDummy)
                    If rc <> 0 Then
                        LogEvent "Errore scrittura DB109.DW" & 2 * (10 * iBil + 4) & " (Flag reset) rc=" & rc, vbLogEventTypeError
                    End If
                    aDummy(1) = lStart(iBil)
                    '03/08 rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 1, 1, aDummy)
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil, 1, aDummy)
                    If rc <> 0 Then            'Start
                        LogEvent "Errore scrittura DB109.DW" & 2 * (10 * iBil) & " (Flag start) rc=" & rc, vbLogEventTypeError
                    End If
                    aDummy(1) = 0
                    rc = mRK512.RK512Put(rkData, 109, 10 * iBil + 2, 1, aDummy)
                    If rc <> 0 Then            'Fine
                        LogEvent "Errore scrittura DB109.DW" & 2 * (10 * iBil + 2) & " (Flag fine dosaggio) rc=" & rc, vbLogEventTypeError
                    End If
                    aDummy(1) = 1
                    rc = mRK512.RK512Put(rkData, 109, 1, 1, aDummy)
                    If rc <> 0 Then                'EOT
                        LogEvent "Errore scrittura DB109.DW2 (EOT) rc=" & rc, vbLogEventTypeError
                    End If
                    RaiseEvent InizioDosaggio(mIdProd, mCurrMisc, sBilName(iBil))
                End If
            Next
            mScanTimer.Enabled = True
        Else
            mrsDettProd.Close
            mrsGiac.Close
            ShowMsg "Fine Produzione nr. " & mIdProd
            If bAbort Then
                Call UpdStato(mIdProd, -2)    'produzione annullata
            Else
                Call UpdStato(mIdProd, 2)    'fine produzione
            End If
            
            RaiseEvent FineProduzione(mIdProd)
            
            If bHidden Then
                Call DeleteProd(mIdProd)
            End If
            
            mIdProd = 0
            mNrMisc = 0
            mCurrMisc = 0
            mTimer.Enabled = True
        End If
    Else
        mScanTimer.Enabled = True
    End If
    Exit Sub

OnRkErr:
    ShowMsg "mScanTimer.Errore rk512: " & Err.Description
    Resume Next
End Sub

Private Sub mTimer_Timer()
    mTimer.Enabled = False
    Call mAvvia
End Sub
Private Sub DeleteProd(IdProd As Long)
Dim cmd As New Command

    cmd.ActiveConnection = gConn
    cmd.CommandText = "DELETE FROM T_Produzioni WHERE IDProd=" & IdProd
    cmd.Execute
    
End Sub
